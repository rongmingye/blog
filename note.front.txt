window.location
	window.location.href; 设置或获取url 为字符串
			ps: alert(window.location.href); 则输出：http://localhost:8086/topic/index?topicId=361

	window.location.search; 设置或获取href属性的问号后面部分
			ps: alert(window.location.search); 则输出：?topicId=361

	window.location.pathname; 设置或获取对象指定的文件名或路径 (问好前面, 端口后面)
			ps: http://localhost:8086/topic/index?topicId=361
				alert(window.location.pathname); 则输出：/topic/index

	window.location.port; 设置或获取与 URL 关联的端口号码。
			ps: alert(window.location.port); 则输出：8086
